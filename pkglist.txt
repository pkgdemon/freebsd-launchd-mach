# One pkg per line; lines starting with # are ignored.
# Runtime packages — installed in the chroot and shipped in the ISO.

# ---- gershwin/GNUstep runtime deps ----

# libs-base configure: GSXML / NSXMLNode / NSPropertyListSerialization
# XML branch. libgnustep-base.so links against libxml2.so at runtime.
libxml2

# libs-base configure: NSLocale / NSString unicode handling.
# libgnustep-base.so links against libicui18n.so + libicuuc.so at runtime.
icu

# Runtime dep surfaced at boot: ld-elf.so.1 fails to load
# libgnustep-base.so because of "Shared object libffi.so.8 not found".
# libffi gets pulled in as a transitive of cmake/ninja at build time but
# removed by `pkg autoremove` after the buildpkgs purge. Pin it
# explicitly here so it stays on the ISO.
libffi

# ---- firmware blobs ----

# GPU firmware blobs metaport — pulls all flavored gpu-firmware-*-kmod
# subpackages.
gpu-firmware-kmod

# WiFi firmware blobs. Drivers themselves are in base.txz; firmware kmods
# are loaded when matching hardware attaches.
wifi-firmware-iwlwifi-kmod
wifi-firmware-rtw88-kmod
wifi-firmware-rtw89-kmod

# ---- hypervisor guest additions ----

virtualbox-ose-additions-nox11
open-vm-tools-nox11

# ---- network ----

# Network auto-config: dhcpcd in master mode.
dhcpcd

# Ports wpa_supplicant supersedes the base version.
wpa_supplicant
